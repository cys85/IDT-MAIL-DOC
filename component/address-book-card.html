# 组件源码

<template>
  <div>
    <TheCard class="address-book-card" :border="true">
      <template v-slot:title>
        <div class="address-book-card__title">{{ title }}({{totalMailNum}})</div>
      </template>
      <div class="address-book-card__body">
        <div class="address-book-card__body__item">
          <p class="address-book-card__body__item__text">有邮件往来</p>
          <div class="address-book-card__body__item__line__warp">
            <div class="address-book-card__body__item__line"
            :style="{width: mailNum*100/(mailNum+noMailNum)+'%'}" ></div>
          </div>
          <span class="address-book-card__body__item__data">{{mailNum}}人</span>
        </div>
        <div class="address-book-card__body__item">
          <p class="address-book-card__body__item__text">无邮件往来</p>
          <div class="address-book-card__body__item__line__warp">
            <div class="address-book-card__body__item__line"
            :style="{width: noMailNum*100/(mailNum+noMailNum)+'%'}" ></div>
          </div>
          <span class="address-book-card__body__item__data">{{noMailNum}}人</span>
        </div>
      </div>

      <template v-slot:footer>
        <div class="address-book-card__footer">
          <AddressBookCardBtns :btns="['browse', 'edit', 'delete']"></AddressBookCardBtns>
        </div>
      </template>
    </TheCard>
  </div>
</template>

<script lang="ts">
/**
 * @author 孙英萌
 * @date 2021-07-28
 * @description 通讯录card组件
 * @prop {number} mailNum 有邮件往来数量,进度条的逻辑 mailNum/(mailNum+noMailNum)
 * @prop {number} noMailNum 无邮件往来数量,进度条的逻辑 noMailNum/(mailNum+noMailNum)
 * @prop {string} meta  元数据
 * @slot name='btns'
 */
import { Component, Vue, Prop } from 'vue-property-decorator';
import TheCard from '@/components/TheCard';
import AddressBookCardBtns from '@/views/AddressBook/components/AddressBookCardBtns.vue';

@Component({
  components: {
    TheCard,
    AddressBookCardBtns,
  },
})
export default class AddressBookCard extends Vue {
  @Prop() title!: string;

  @Prop() mailNum!: number;

  @Prop() noMailNum!: number;

  @Prop() totalMailNum!: number;

  @Prop() meta!: string;

  created() {
    this.totalMailNum = (this.noMailNum + this.mailNum);
  }
}
</script>
