# 组件源码

<template>
  <div class="mail-search-form-view">
    <div class="mail-search-form-view__header">
      <div class="mail-search-form-view__title mail-search-form-view__keywords">关键字：</div>
      <div class="mail-search-form-view__keywordsInput">
        <el-input v-model="keywordsInput" placeholder="请输入关键字"></el-input>
      </div>
    </div>
    <div class="mail-search-form-view__tips">多个关键字请用英文状态下的逗号分隔</div>
    <div class="mail-search-form-view__title mail-search-form-view__searchNode">搜索节点：</div>
    <div class="mail-search-form-view__title mail-search-form-view__searchRange">搜索范围：</div>
    <div class="mail-search-form-view__title mail-search-form-view__timeRange">时间范围：</div>
    <div class="mail-search-form-view__searchNodeCheckbox">
      <el-checkbox-group v-model="searchNode">
        <el-checkbox v-for="(item,index) in searchNodeType" :key="index" :label="item">
        </el-checkbox>
      </el-checkbox-group>
    </div>
    <div class="mail-search-form-view__searchRangeCheckbox">
      <el-checkbox-group v-model="searchRange">
        <el-checkbox v-for="(item,index) in searchRangeType" :key="index" :label="item">
        </el-checkbox>
      </el-checkbox-group>
    </div>
    <div class="mail-search-form-view__timeHorizon">
      <time-horizon @outputTime="outputTime" ref="childRef"></time-horizon>
    </div>
    <div class="mail-search-form-view__formItemDescription">
        <form-item-description text="时间范围设置规则是起始时间的0点开始到结束时间的23点59分59秒" />
    </div>
    <div class="mail-search-form-view__btnGroup">
      <div class="mail-search-form-view__btn1">
        <el-button type="primary" @click="submit">搜索</el-button>
      </div>
      <div class="mail-search-form-view__btn2">
        <el-button @click="reset">重置</el-button>
      </div>
    </div>
  </div>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-30
 * 说明：邮件搜索表单
 * Props：无
 * Emit：submit（）传出form数据
 *       cancel()点击重置后抛出的事件
 */
import { Component, Vue } from 'vue-property-decorator';
import TimeHorizon from '@/components/TimeHorizon';
import FormItemDescription from '@/components/FormItemDescription';

@Component({
  components: {
    TimeHorizon,
    FormItemDescription,
  },
})
export default class MailSearchFormView extends Vue {
  keywordsInput = ''

  searchNode = ['标题', '收件人', '内容', '附件']

  searchNodeType = ['标题', '收件人', '内容', '附件']

  searchRange = ['收件箱', '发件箱', '草稿箱', '废件箱', '收藏附件', '工作邮件', '生活邮件']

  searchRangeType = ['收件箱', '发件箱', '草稿箱', '废件箱', '收藏附件', '工作邮件', '生活邮件']

  startTime = ''

  endTime = ''

  outputTime(startTime:string, endTime:string) {
    this.startTime = startTime;
    this.endTime = endTime;
  }

  submit() {
    this.$emit('submit', {
      startTime: this.startTime,
      endTime: this.endTime,
      keywords: this.keywordsInput,
      searchNode: this.searchNode,
      searchRange: this.searchRange,
    });
  }

  reset() {
    this.startTime = '';
    this.endTime = '';
    this.keywordsInput = '';
    this.searchNode = this.searchNodeType;
    this.searchRange = this.searchRangeType;
    const thisRefs:any = this.$refs;
    thisRefs.childRef.reset();
    this.$emit('cancel');
  }
}
</script>
