# 组件源码

<template>
<div>
  <div class="address-book-card-btns" v-for="(item,index) in btns" :key="index">
    <el-button type="text"  @click="click">{{btn[item]}}</el-button>
  </div>
</div>
</template>

<script lang="ts">
/**
 * @author 孙英萌
 * @date 2021-07-28
 * @description 通讯录card按钮组组件
 * @prop {string[]} btns, 数据
   @prop {string} btns[].browse 查看
   @prop {string} btns[].edit  编辑
   @prop {string} btns[].delete  删除
   @prop {string} meta 元数据
   @emit click callback meta
                            如果数组中包含对应的按钮则有相应的点击事件
                            browse：callback参数 meta，点击查看按钮触发
                            edit：callback参数 meta ，点击编辑按钮触发
                            delete：callback参数 meta，点击删除按钮触发
 */
import {
  Prop, Component, Vue,
} from 'vue-property-decorator';

@Component({})
export default class AddressBookCardBtns extends Vue {
  @Prop() btns!:string[];

  @Prop() meta!:string

  btn = {
    browse: '查看',
    edit: '编辑',
    delete: '删除',
  }

  click() {
    return this.meta;
  }
}
</script>
