# 组件源码

<template>
  <div class="mail-capacity">
    <div class="mail-capacity__text">邮箱容量 <span>{{used}}/{{total}}</span></div>

    <div class="mail-capacity__bar">
      <div class="mail-capacity__percent"
          :class="{['mail-capacity__percent--warn']:percent>=90}"
          :style="{width:percent+'%'}"></div>
    </div>
  </div>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-22
 * 修改 2021-07-28
 * 说明：邮箱容量组件改
 * 参数：totalCapacity（number）:总容量 kb
 *      percent（number）：所占容量百分比如：取值（0 ~ 100）
 */
import { Component, Prop, Vue } from 'vue-property-decorator';

@Component({
})
export default class MailCapacity extends Vue {
  @Prop()totalCapacity!:number

  @Prop()percent!:number

  total = ''

  used = ''

  created() {
    this.total = this.normalizeCapacity(this.totalCapacity);
    this.used = this.normalizeCapacity(this.totalCapacity * (this.percent / 100));
  }

  normalizeCapacity(capacity:number):string {
    // 输入任意整数（kb）返回对应kb，mb，或gb，保留两位小数
    if (capacity < 1024) {
      return `${capacity}KB`;
    }
    if (capacity <= 1024 ** 2) {
      return `${parseFloat((capacity / 1024).toFixed(2))}MB`;
    }
    return `${parseFloat((capacity / 1024 ** 2).toFixed(2))}GB`;
  }
}

</script>
