# 组件源码

<template>
  <div class="password-settings">
    <the-card title="密码修改">
      <template>
        <div class="password-settings__body">
          <el-form label-position="left" label-width="100px" :model="passwordForm" :rules="rules"
                  ref="passwordForm" :disabled="formLoading"
                  :hide-required-asterisk="true">
            <el-form-item label="原密码:" prop="old">
                <el-input v-model="passwordForm.old"></el-input>
            </el-form-item>
            <el-form-item label="新密码:"  prop="newPassword">
                <el-input v-model="passwordForm.newPassword"
                          type="password"></el-input>
            </el-form-item>
              <el-form-item label="确认密码:"  prop="newAgain">
                <el-input v-model="passwordForm.newAgain"
                          type="password"></el-input>
              </el-form-item>
            </el-form>
        </div>
      </template>

      <template v-slot:footer>
        <div class="password-settings__footer">
            <el-button type="primary" :loading="formLoading" @click="submitPassword">
              {{formLoading?'加载中':'确定'}}
            </el-button>
            <el-button :disabled="formLoading" @click="reset">取消</el-button>
        </div>
      </template>
    </the-card>
  </div>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-27
 * 说明：密码设置组件
 * Props：old-password：原密码
          new-password：新密码
          new-paddword-again：确认密码
          loading：true | false，true时候表单不允许修改，“确认”按钮变为提交中，“取消”按钮表位disable状态
 * Emit：submit：callback参数：old-password、new-password、new-paddword-again
 */
import {
  Component, Vue, Prop, Watch,
} from 'vue-property-decorator';
import TheCard from '@/components/TheCard';

@Component({
  components: {
    TheCard,
  },
})
export default class PasswordSetting extends Vue {
  @Prop()oldPassword!:string

  @Prop()newPassword!:string

  @Prop()newPasswordAgain!:string

  @Prop()loading!:boolean

  passwordForm = {
    old: '',
    newPassword: '',
    newAgain: '',
  }

  formLoading=false

  checkPassword(rule:Record<string, unknown>, value:string, callback:any) {
    const { newPassword } = this.passwordForm;
    if (value === '') {
      console.log('值为空');
      callback(new Error('请确认新密码'));
    } else if (value !== newPassword) {
      callback(new Error('两次密码输入不一致，请重新输入'));
    } else {
      callback();
    }
  }

  rules={
    old: [{ required: true, message: '原密码不允许为空', trigger: 'blur' }],
    new: [{ required: true, message: '请输入新密码', trigger: 'blur' }],
    newAgain: [{ validator: this.checkPassword, trigger: 'blur' }],
  }

  // todo:表单验证需要更改
  submitPassword() {
    const ref:any = this.$refs.passwordForm;
    ref.validate((valid:boolean) => {
      if (valid) {
        this.$emit('submit', this.passwordForm.old, this.passwordForm.newPassword, this.passwordForm.newAgain);
      }
    });
  }

  reset() {
    this.passwordForm = {
      old: this.oldPassword,
      newPassword: this.newPassword,
      newAgain: this.newPasswordAgain,
    };
  }

  created() {
    this.formLoading = this.loading;
    this.passwordForm.old = this.oldPassword;
    this.passwordForm.newPassword = this.newPassword;
    this.passwordForm.newAgain = this.newPasswordAgain;
  }

  @Watch('loading')
  watchLoaing() {
    this.formLoading = this.loading;
  }
}
</script>
