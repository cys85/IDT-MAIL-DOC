# 组件源码

<template>
  <div class="mail-card" @click="mailCardClick" :class="{bgActive:isActived}">
    <div class="mail-card__point" :style="{background:isUnRead?'#f00':'#CAC0C0'}"></div>
    <img class="mail-card__head"  :src="avatar">
    <div class="mail-card__name" :class="{textActive:isActived}">{{name}}</div>
    <img class="mail-card__star" :src="require(`@/assets/images/icon/star.png`)"
    v-if="isStarMan">
    <div class="mail-card__date" :class="{textActive:isActived}">{{date}}</div>
    <div class="mail-card__title" :class="{textActive:isActived}">{{returnTitle}}</div>
  </div>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-23
 * 说明：邮件卡片组件
 * 参数：isActived（boolean）：是否激活状态
 *     ：isUnRead（boolean）：已读未读
 *     ：isStarMan（boolean）：是否星标人
 *     ：name（string）：姓名
 *     ：title（string）：邮件标题
 *     ：date（string）：日期
 *     ：mete（string）：元数据
 *     ：avatar（string）：头像地址：支持： require(`@/assets/images/icon/notRead.png`)
 */
import {
  Component, Vue, Prop, Emit,
} from 'vue-property-decorator';

@Component({
})
export default class MailCard extends Vue {
  @Prop({ default: false })isActived!:boolean

  @Prop({ default: false })isUnRead!:boolean

  @Prop({ default: false })isStarMan!:boolean

  @Prop()name!:string

  @Prop()title!:string

  @Prop()date!:string

  @Prop()mete!:string

  @Prop()avatar!:string

  @Emit('on-click')
  mailCardClick():string {
    return this.mete;
  }

  get returnTitle():string {
    const max = 25;
    return this.title.length > max ? `${this.title.substr(0, max)}...` : this.title;
  }
}
</script>
