# 组件源码

<template>
<div class="time-horizon"  :class="{['time-horizon--small']:size === 'small'}">
 <el-date-picker
      v-model="time"
      type="daterange"
      range-separator="到"
      :unlink-panels="true"
      :clearable="true"
      :default-time="['00:00:00', '23:59:59']"
      start-placeholder="开始日期"
      end-placeholder="结束日期">
    </el-date-picker>
    <div class="time-horizon__out">
      <time-horizon-output @outputTime="resetTime" :size="size"></time-horizon-output>
    </div>
</div>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-29
 * 说明：时间范围组件（样式待更改）
 * Props：无
 * Emit：outputTime（startTime，endTime）时间改变自动触发
 */
import {
  Component, Vue, Watch, Prop,
} from 'vue-property-decorator';

import TimeHorizonOutput from './TimeHorizonOutput.vue';

@Component({
  components: {
    TimeHorizonOutput,
  },
})
export default class TimeHorizon extends Vue {
  @Prop({ default: true })clear!:boolean

  @Prop({
    default: 'normal',
    validator(value) {
      return ['normal', 'small'].includes(value);
    },
  })size!:string

  time = []

  resetTime(start:never, end:never) {
    this.time = [start, end];
  }

  @Watch('time')
  watchTime() {
    if (this.time === null) return;
    this.$emit('outputTime', this.time[0], this.time[1]);
  }

  reset() {
    this.time = [];
  }
}
</script>
