# 组件源码

<template>
  <div class="send-and-receive-time">
    <img
      class="send-and-receive-time-icon"
      :src="require(`@/assets/images/icon/${icon}.png`)"
      alt="time-icon"
    />
    {{ text }} ：{{ time }}
  </div>
</template>

<script lang="ts">
/**
 * @author 孙英萌
 * @description 收发时间组件
 * @prop {string} time 收发时间
 * @prop {string} type send | receive；
                       send：代表发件，receive：代表收件
 */
import { Component, Prop, Vue } from 'vue-property-decorator';

@Component({})
export default class SendAndReceiveTime extends Vue {
  @Prop({
    validator(value) {
      if (value === 'send' || value === 'receive') {
        return true;
      }
      return false;
    },
  })
  type!: string;

  @Prop() time!: string;

  get text(): string {
    return this.type === 'send' ? '发件时间' : '收件时间';
  }

  get icon(): string {
    return this.type === 'send' ? 'time-yellow' : 'time-blue';
  }
}
</script>
