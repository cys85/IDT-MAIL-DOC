# 组件源码

<template>
  <div class="custom-folder">
    <div class="custom-folder-header">
      <div class="custom-folder-header__downArrow"></div>
        <img class="custom-folder-header__downArrowImg"
            :class="{['custom-folder-header__downArrowImg--collapsed']:collapsedStatus}"
            :src="require('@/assets/images/icon/down-arrow.png')">
        <div class="custom-folder-header__title" @click="changeCollapsed">其他文件夹</div>
        <div class="custom-folder-header__slot1"><slot name="slot1"/></div>
        <div class="custom-folder-header__slot2"><slot name="slot2"/></div>
    </div>
    <div   v-show="!collapsedStatus">
      <CustomFolderItem v-for="(item,index) in folders"
      :name="item.name"
      :meta="item.meta"
      :key="index"
      @customFolderItemClick="output"/>
    </div>

  </div>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-26
 * 说明：CustomFolder组件，子组件（CustomFolderItem）
 * Props：folders（[{name://问件夹名称,meta://元数据}]）：文件夹列表
 *        collapsed（boolean）:是否折叠状态，true表示折叠
 * Emit：onClickFolder:传出（文件夹名称，元数据）
 */
import {
  Component, Vue, Prop,
} from 'vue-property-decorator';
import CustomFolderItem from './CustomFolderItem.vue';

@Component({
  components: {
    CustomFolderItem,
  },
})
export default class CustomFolder extends Vue {
  @Prop()folders!:Array<Record<string, unknown>>

  @Prop({ default: false })collapsed!:boolean

  collapsedStatus = true

  created() {
    this.collapsedStatus = this.collapsed;
  }

  output(data:Record<string, unknown>) {
    this.$emit('on-click-folder', data.name, data.meta);
  }

  changeCollapsed() {
    this.collapsedStatus = !this.collapsedStatus;
  }
}
</script>
