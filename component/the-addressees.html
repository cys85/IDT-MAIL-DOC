# 组件源码

<template>
  <div class="the-addressees">
    <div :id="uid+'AddressTitle'" class="the-addressees-title" :style="{width:width+'px'}">
          <div id="chen">
           <div class="the-addressees-title-contnet the-addressees-title-show"
           v-for="(item,index) in data" :key="index"
                :style="{color:item.isUnRead===true?'#ccc':'#3377ff'}">
             {{item.name+'，'}}</div>
          </div>
      <el-popover
        class="the-addressees-pop"
        placement="bottom-end"
        width="500"
        trigger="click"
        v-if="show"
        >
          <img slot="reference" class="the-addressees-title-icon"
               src="@/assets/images/icon/down.png" alt="" />
    <div class="the-addressees-title-contnet" v-for="(item,index) in popoverData" :key="index"
                :style="{color:item.isUnRead===true?'#ccc':'#3377ff'}">
             {{item.name+'，'}}
        </div>
      </el-popover>
    </div>
  </div>
</template>
<script lang="ts">
/**
 * @author 孙英萌
 * @description 收件人集合组件
 * @prop {string} width  组件宽度 可以自行输入
 * @prop {Object[]} data
 * @prop {string} data[].name: 姓名
 * @prop {boolean} data[].isUnRead: 是否未读，true为已读为灰色字体，false为未读为蓝色字体
 */
import {
  Prop, Component, Vue, Watch,
} from 'vue-property-decorator';

@Component({})
export default class TheAddressees extends Vue {
  @Prop() width!: string;

  @Prop() data!: Record<string, unknown>[];

  @Prop() isUnRead!:boolean

  @Prop() name!:string

  uid = ''

  show=true

  popoverData = []

  created() {
    this.uid = this.$vnode.tag || 'the-addressees';
  }

  @Watch('data', { immediate: true, deep: true })
  dataWatch() {
    this.$nextTick(() => {
      const doms = document.getElementsByClassName('the-addressees-title-show');
      let totalWidth = 0;
      let maxIndex = 0;
      for (let i = 0; i < doms.length; i += 1) {
        totalWidth += (doms[i] as any).offsetWidth;
        if (totalWidth > (document as any).getElementById(`${this.uid}AddressTitle`).offsetWidth) {
          maxIndex = i;
          break;
        }
      }
      this.popoverData = JSON.parse(JSON.stringify(this.data));
      this.popoverData.splice(0, maxIndex);
      if (totalWidth < (document as any).getElementById(`${this.uid}AddressTitle`).offsetWidth) {
        this.popoverData = [];
        this.show = false;
      }
    });
  }
}
</script>
