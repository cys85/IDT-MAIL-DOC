  </div>

/**
 * @author 孙英萌
 * @description 分类tab组件2

 * @prop {string} meta 元数据
 * @emit  click callback meta
 */

  Prop, Component, Vue, Emit,
} from 'vue-property-decorator';

@Component({})
export default class ClassType2 extends Vue {
  @Prop() number!: number;

  @Prop() text!: string;

  @Prop() meta!: string;

  @Emit('submit')
  click() {
    // console.log(123);
    return this.meta;
  }
}
</script>
