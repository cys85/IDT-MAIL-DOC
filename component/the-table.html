# 组件源码

<template>
  <div class="the-table">
    <el-table
      size="large"
      stripe
      v-bind="$attrs" v-on="$listeners"
      style="width: 100%">
      <slot />
    </el-table>
    <the-pagination
      class="the-table__pagination"
      :default-current="defaultCurrent"
      :default-page-size="defaultPageSize"
      v-on="$listeners"
      :total="total" />
  </div>
</template>
<script lang="tsx">
/* eslint-disable no-param-reassign */
/**
 * 作者：陈雨声
 * 时间：2021-07-25
 * 说明：element table的二次封装
 * Props：继承element table所有props
 * Emit：继承element table所有emit
 */
import { Component, Prop, Vue } from 'vue-property-decorator';
import ThePagination from '@/components/ThePagination';

@Component({
  components: {
    ThePagination,
  },
})
export default class TheTable extends Vue {
  @Prop(
    {
      type: Number,
    },
  )
  defaultCurrent!: number // 当前第几页

  @Prop(
    {
      type: Number,
    },
  )
  total!: number // 总数

  @Prop(
    {
      type: Number,
    },
  )
  defaultPageSize!: number // 每页大小
}
</script>
