# 组件源码

<template>
  <div class="execution-cycle">
    <div class="execution-cycle__word1">每</div>
    <div class="execution-cycle__input">
      <el-input v-model="form.num"></el-input>
    </div>
    <div class="execution-cycle__select">
      <el-select v-model="form.unit" placeholder="请选择">
        <el-option v-for="item in units" :key="item.value" :label="item.label" :value="item.value">
        </el-option>
      </el-select>
    </div>
    <div class="execution-cycle__word2">执行一次</div>
  </div>
</template>

<script lang="ts">
/**
 * @author 陈政
 * @date 2021-08-04
 * @description 执行循环组件
 * @prop { object } value - 缺省值 支持v-model
 * @prop { number } value.num - 数量
 * @prop { string } value.unit - 单位
 * @prop { object[] } units - 备选单位集合
 * @prop { string } units[].label - 备选单位显示值
 * @prop { string } units[].value - 备选单位实际值
 * @prop { boolean } isDefault - 缺省选中 true：为缺省选中，缺失或false为非缺省
 * @event
 */
import { Component, Prop, Vue } from 'vue-property-decorator';

@Component({})
export default class ExecutionCycle extends Vue {
  @Prop({
    default() {
      return [{ label: '天', value: 'day' }, { label: '月', value: 'month' }];
    },
  }) units!: Array<Record<string, unknown>>;

  @Prop() value!: Record<string, unknown>;

  form = {};

  created() {
    this.form = this.value;
  }
}
</script>
