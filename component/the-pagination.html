# 组件源码

<template>
  <div class="the-pagination">
    <div class="the-pagination__info">
      共 {{total}} 条记录 第 <span>{{defaultCurrent}}</span> / {{pageTotal}} 页
    </div>
    <el-pagination
      background
      layout="prev, pager, next, jumper"
      :total="total"
      @current-change="currentChange">
    </el-pagination>
  </div>
</template>

<script lang="ts">
/**
 * 作者：陈雨声
 * 时间：2021-07-25
 * 说明：基于a-pagination二次包裹封装
 * Props：
 * Emit：
 */
import {
  Component, Emit, Prop, Vue,
} from 'vue-property-decorator';

@Component({
})
export default class ThePagination extends Vue {
  // 当前第几页
  @Prop({
    type: Number,
    default: 0,
  }) defaultCurrent !: number

  // 总数
  @Prop({
    type: Number,
    default: 0,
  }) total !: number

  @Prop({
    type: Number,
    default: 10,
  })
  defaultPageSize !:number

  get pageTotal(): number {
    return this.total === 0 ? 0 : Math.ceil(this.total / this.defaultPageSize);
  }

  @Emit('currentChange')
  currentChange(currentPages: number) {
    return currentPages;
  }
}
</script>
