# 组件源码

<template>
  <div class="the-card">
    <div class="the-card__header">
      <slot name="title"><div class="the-card__title">{{title}}</div></slot>
    </div>
    <div class="the-card__body">
      <slot name="default"></slot>
    </div>
    <div class="the-card__border" v-if="border"></div>
    <div class="the-card__footer">
      <slot name="footer"></slot>
    </div>
  </div>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-28
 * 说明：theCard组件，边框
 * Props：title（string）
 *      ：border（boolean）
 * Emit：
 */
import { Component, Vue, Prop } from 'vue-property-decorator';

@Component({
})
export default class TheCard extends Vue {
  @Prop()title!:string

  @Prop({ default: false })border!:boolean
}
</script>
