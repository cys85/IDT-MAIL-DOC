# 组件源码

<template>
  <div class="personal-information-settings">
    <TheCard title="基本信息">
      <template>
        <div class="personal-information-settings__body">
          <el-form label-position="left" label-width="80px" :disabled="loading"
                  :hide-required-asterisk="true">
            <el-form-item label="部门:">
                <el-input v-model="dept" disabled></el-input>
            </el-form-item>
            <el-form-item label="姓名:">
                <el-input v-model="name"   disabled></el-input>
            </el-form-item>
              <el-form-item label="手机号:">
                <el-input v-model="inputPhoneNumber"></el-input>
              </el-form-item>
            </el-form>
        </div>
      </template>
      <template v-slot:footer>
        <div class="personal-information-settings__footer">
            <el-button type="primary" :loading="loading" @click="submitForm">
              {{loading?'加载中':'确定'}}
            </el-button>
            <el-button @click="reset">取消</el-button>
        </div>
      </template>
    </TheCard>
  </div>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-27
 * 说明：PersonalInformationSettings组件
 * Props：dept: 部门，不可修改
          name：姓名，不可修改
          phone-number：手机号
          loading：true | false ，true时，手机号不允许修改
 * Emit：submit：callback参数：phone-number
        点击取消按钮，重置表单（重置为修改前的值）
 */
import {
  Component, Prop, Vue, Emit,
} from 'vue-property-decorator';
import TheCard from '@/components/TheCard';

@Component({
  components: {
    TheCard,
  },
})
export default class PersonalInformationSettings extends Vue {
  @Prop()dept!:string

  @Prop()name!:string

  @Prop()phoneNumber!:string

  @Prop()loading!:boolean

  inputDept=''

  inputName=''

  inputPhoneNumber=''

  inputLoading=false

  created() {
    this.inputDept = this.dept;
    this.inputName = this.name;
    this.inputPhoneNumber = this.phoneNumber;
    this.inputLoading = this.loading;
  }

@Emit('submit')
  submitForm() {
    return this.inputPhoneNumber;
  }

reset() {
  this.inputPhoneNumber = this.phoneNumber;
}
}
</script>
