# 组件源码

<template>
  <div>
    <tree-transfer :title="['全部联系人','该组联系人']" :from_data='personList' :to_data='res' mode="transfer"
    :button_text = "['移入','移出']"
     @add-btn='add' @remove-btn='remove'
     height='370px' width='630px' filter openAll>
    </tree-transfer>
  </div>
</template>

<script lang="ts">
/**
 * @author 陈政
 * @date 2021-08-03
 * @description：人员选择穿梭树
 * @param personList
 * @param
 * @event output:(list)=>list(更改自动抛出list)
 */
import { Component, Vue, Prop } from 'vue-property-decorator';
import treeTransfer from 'el-tree-transfer';

@Component({
  components: {
    treeTransfer,
  },
})
export default class UserTransferSelect2 extends Vue {
  @Prop()personList!:Array<Record<string, unknown>>

  res=[]

  add() {
    this.$emit('output', this.res);
  }

  remove() {
    this.$emit('output', this.res);
  }
}
</script>
