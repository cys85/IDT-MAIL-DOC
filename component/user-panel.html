# 组件源码

<template>
  <el-dropdown  class="user-panel">
    <div class="user-panel__body">
      <el-avatar :size="32" :src="avatarUrl" class="user-panel__user" />
      <p class="user-panel__userStr">{{ userName }}</p>
      <img src="@/assets/images/down.png" class="user-panel__down" />
    </div>
    <el-dropdown-menu class="user-panel__dropdown" slot="dropdown">
      <el-dropdown-item @click.native="systemSettings">
        <the-icon type="icon-shezhi" /> 系统设置
      </el-dropdown-item>
      <el-dropdown-item divided @click.native="cancel">
        <the-icon type="icon-tuichu" /> 退出
      </el-dropdown-item>
    </el-dropdown-menu>
  </el-dropdown>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-18
 * 说明：用户面板
 * 参数：
 * @param {string} userName - 用户名
 * @param {boolean} avatarUrl - 用户头像地址
 * @event cancel: void - 【退出按钮】点击事件
 * @event system-settings: void - 【系统设置】按钮事件
 */
import {
  Component, Vue, Prop, Emit,
} from 'vue-property-decorator';
import TheIcon from '@/components/TheIcon';

@Component({
  components: {
    TheIcon,
  },
})
export default class UserPanel extends Vue {
  @Prop() userName!: string;

  @Prop() avatarUrl!: string;

  @Emit()
  cancel() {}

  @Emit()
  systemSettings() {}
}
</script>
