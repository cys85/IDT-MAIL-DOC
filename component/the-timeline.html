# TheTimeline

<template>
  <div class="the-timeline">
        <slot></slot>
  </div>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-29
 * 说明：时间线列表边框
 * Props：slot
 */
import { Component, Vue } from 'vue-property-decorator';

@Component({
  provide() {
    return {
      timeline: this,
    };
  },
})
export default class TheTimeline extends Vue {
  out(meta:string, date:string) {
    this.$emit('clickItem', meta, date);
  }
}
</script>

# TimelineItem

<template>
  <div class="time-line-item">
    <div class="time-line-item__header">
      <div class="time-line-item__headerDate">{{dateObj.day}}</div>
      <div class="time-line-item__headerYear">{{dateObj.year}}</div>
    </div>
    <div class="time-line-item__circle"></div>
    <div class="time-line-item__line"></div>
    <div class="time-line-item__body" @click="itemClick">
      <slot></slot>
    </div>
  </div>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-29
 * 说明：时间线列表子元素组件
 * Props：date: 时间 (2021-07-29)标准格式
          slot
          meta: 元数据
 * Emit：无
 */
import { Component, Vue, Prop } from 'vue-property-decorator';

@Component({
  inject: ['timeline'],
})
export default class TimelineItem extends Vue {
  @Prop()date!:string

  @Prop()meta!:string

  get dateObj() :Record<string, unknown> {
    const dateStr = this.date;
    const year = dateStr.substr(0, 4);
    const day = dateStr.substr(5, 9);
    return { year, day };
  }

  itemClick() {
    const self: any = this;
    self.timeline.out(this.meta, this.date);
  }
}
</script>
