# 组件源码

<template>
  <div class="login-form">
    <div class="login-form__border">
      <div class="login-form__title">账户密码登录</div>
    </div>
    <el-form :rules="rules" :model="form" :disabled="loading" ref="loginForm">
    <div class="login-form__input1">
        <el-form-item  prop="userName">
        <el-input placeholder="请输入您的账号" v-model="form.userName" :rules="rules">
            <div slot="prefix">
              <img :src="require('@/assets/images/icon/login-form-user.png')">
            </div>
        </el-input>
        </el-form-item>
    </div>
    <div class="login-form__input2">
        <el-form-item  prop="password">
          <!-- TODO: 密码输出不能是明文的， 陈雨声已经修正 加入 type="password" autocomplete="off" -->
        <el-input placeholder="请输入您的密码" v-model="form.password" type="password" autocomplete="off"
                  @keyup.enter.native="login">
           <div slot="prefix">
              <img :src="require('@/assets/images/icon/login-form-password.png')">
            </div>
        </el-input>
        </el-form-item>
    </div>
    </el-form>
    <div class="login-form__slidingBlock"></div>
    <div class="login-form__forgotPassword">
    <el-tooltip effect="dark" content="请与管理员联系" placement="bottom">
      <span @click="forgotPassword" style="cursor:pointer;">忘记密码？</span>
    </el-tooltip>
    </div>
    <div class="login-form__submit">
      <el-button :loading="loading" @click="login">{{loading?'登录中...':'登录'}}</el-button>
    </div>

  </div>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-27
 * 说明：登录表单组件
 * Props：userName: string
          password: string
          loading：true | false ，如果为true 登录按钮名称变为 “登录中...”
 * Emit：login：callback参数：userName、password
 */
import { Component, Vue, Prop } from 'vue-property-decorator';

@Component({
})
export default class LoginForm extends Vue {
  @Prop()userName!:string

  @Prop()password!:string

  @Prop()loading!:boolean

  // inputLoading=false

  form={ userName: '', password: '' }

   rules={
     userName: [{ required: true, message: '请输入您的帐号', trigger: 'blur' }],
     password: [{ required: true, message: '请输入您的密码', trigger: 'blur' }],
   }

   created() {
     // TODO: 为什么要把外部传入的loading赋值给 inputLoading？
     // TODO: loading如果更新后，inputLoading 没有对应更新
     //  this.inputLoading = this.loading;
     this.form.userName = this.userName;
     this.form.password = this.password;
   }

   login() {
     const ref:any = this.$refs.loginForm;
     ref.validate((valid:boolean) => {
       if (valid) {
         this.$emit('login', this.form.userName, this.form.password);
       }
     });
   }

   forgotPassword() {
     console.log('忘记密码');
   }
}
</script>
