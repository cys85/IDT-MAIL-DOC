<template>

    共<span class="page-record-total">{{totalNumber}}</span>条记录&emsp;
    第<span class="page-record-current">{{currentPage}}</span>/{{totalPage}}页
  </div>
</template>

<script lang="ts">
/**
 * 作者：陈政
 * 时间：2021-07-22
 * 说明：页数记录组件
 * 参数：totalNumber:（number） 记录总数
 *       totalPage:(number)总页数
 *       currentPage:(number)当前页数
 */
import { Component, Prop, Vue } from 'vue-property-decorator';

@Component({
})
export default class PageRecord extends Vue {
  @Prop()totalNumber!:number

  @Prop()totalPage!:number

  @Prop()currentPage!:number
}
</script>
