# 组件源码

<template>
  <div class="folder-checkbox-group">
    <el-checkbox  class="folder-checkbox-group-title"
    :indeterminate="isIndeterminate" v-model="checkAll"
    @change="handleCheckAllChange">文件夹名称</el-checkbox>
  <el-checkbox-group v-model="checkedopt" @change="handleCheckedoptChange">
    <el-checkbox class="folder-checkbox-group-items"
                 v-for="(item,index) in opt" :label="item.label" :key="index"
                 v-model="opt[index].value">
                 {{item.label}}
    </el-checkbox>
  </el-checkbox-group>
    <div class="folder-checkbox-group-footer">
      <el-button class="folder-checkbox-group-footer-submit" type="primary" @click="submit"
        >确定</el-button
      >
      <el-button class="folder-checkbox-group-footer-cancel" @click="cancel">取消</el-button>
    </div>
  </div>

</template>

<script lang="ts">
/**
 * @author 孙英萌
 * @description 文件夹选择器组件
 * @prop {any} options：Object[]
   @prop {string} Object[].label 显示名
   @prop {string} Object[].value 值
   @prop {string} Object[].meta  元数据
 * @emit submit:返回选中的value
 * @emit cancel:返回参数无
 */
import { Prop, Component, Vue } from 'vue-property-decorator';

@Component({})
export default class FolderCheckboxGroup extends Vue {
  @Prop() options!:any[]

  @Prop() value!:never[]

  checkAll=false

  checkedopt=['']

  checkedvalue = this.options.map((obj) => obj.value)

  opt=this.options

  isIndeterminate=false

  handleCheckAllChange(value:any) {
    this.checkedopt = value ? this.checkedvalue : [];
    this.isIndeterminate = false;
  }

  handleCheckedoptChange(value:any) {
    const checkedCount = value.length;
    this.checkAll = checkedCount === this.opt.length;
    this.isIndeterminate = checkedCount > 0 && checkedCount < this.opt.length;
  }

  submit() {
    console.log('确定');
    console.log(this.checkedopt);
    return this.opt;
  }

  cancel() {
    console.log('取消');
  }

  created() {
    this.opt = this.options;
    this.checkedopt = this.value;
  }
}
</script>
