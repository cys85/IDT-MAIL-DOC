# 环境变量配置

# .env 通用环境变量

#
# #基础环境变量
#

#
# App
#

# 系统名称
VUE_APP_NAME = '倚天安全邮局'


# 系统版本
VUE_APP_VS = '2.0'
# 基础默认皮肤
VUE_APP_COMPONENT_THEME= 'default'

#
# Router
#

VUE_APP_ROUTER_BASE = '/IDT-Email'

# 路由模式
VUE_APP_ROUTER_MODE = ''

#
# Ajax
#

# API的基础URL
VUE_APP_BASE_API_URL = '/idtAppServiceV6'
# ajax请求超时时间
VUE_APP_AJAX_TIMEOUT = 40000


#
# Other
#

# 全局通用时间格式
VUE_APP_DATE_PATTERN = 'YYYY-MM-DD HH:mm'
